class MySemanticProvider implements vscode.DocumentSemanticTokensProvider {
  async provideDocumentSemanticTokens(document: vscode.TextDocument) {
    const builder = new vscode.SemanticTokensBuilder(legend);
    const text = document.getText();

    // 1. 假设你已经逻辑定位到了目标字符串
    // 2. 调用你的 AST 转换函数
    const myAst = yourParser.parse(text); 

    // 3. 遍历 AST
    this.traverseAst(myAst, (node) => {
      // 假设 node 包含 line, startChar, length 信息
      builder.push(
        node.line, 
        node.startChar, 
        node.length, 
        this.getTokenTypeIndex(node.type), // 对应上面 tokenTypes 的索引
        0 // 修饰符索引
      );
    });

    return builder.build();
  }
}
